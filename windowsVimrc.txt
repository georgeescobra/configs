      au BufReadPost * if line("'\"") > 0 && line("'\"") <= line("$") | exe "normal! g`\"" | endif

" CUSTOM KEY BINDS
    " PANE MOVEMENT
    nnoremap <silent> <Tab>s :vs<CR>:wincmd l<CR>
    nnoremap <silent> <Tab>d :sp<CR>:wincmd k<CR>
    nnoremap <silent> <Tab>h :wincmd h<CR>
    nnoremap <silent> <Tab>l :wincmd l<CR>
    nnoremap <silent> <Tab>j :wincmd j<CR>
    nnoremap <silent> <Tab>k :wincmd k<CR>

    nnoremap <silent> <Tab>f :FZF<CR>

    " writes all the panes
    nnoremap <silent> <Tab>wa :wa<CR>
    nnoremap <silent> <Tab>w :w<CR>
    nnoremap <silent> <Tab>q :q<CR>
    nnoremap <silent> <Tab>1 :q!<CR>
    " This is to go to last line you working on
    nnoremap <silent> <Space>t `" 
    nnoremap <silent> <CR>m :w<CR>:!printf "\e[1;31m\t~~~~~~~~~~~~~new compile~~~~~~~~~~~~~\n\e[0m" && make && ./main<CR>
    nnoremap <silent> <CR>p :w<CR>:!printf "\e[1;31m\t~~~~~~~~~~~~~new compile~~~~~~~~~~~~~\n\e[0m" && python3 %
    nnoremap <CR>1 :!

    nnoremap <Tab>e :edit 
    " Easier Movement Just replacing tab with shift
    nnoremap <Space>j <c-D>
    nnoremap <Space>k <c-U>

    " Moving between buffers
    " Moving between ctags
    nnoremap <Tab>] <c-]>
    nnoremap <Tab>[ <c-[>

    nnoremap <c-m> ma
    nnoremap <c-n> `a

    " Goes to top, bottom of screen
    nnoremap <Space>h <S-h>
    nnoremap <Space>l <S-l>

    nnoremap <Space>f <c-f>
    nnoremap <Space>d <c-b>


    " Moves screen down, up 1
    " Centers cursor middle
    nnoremap <Space>n <c-e>
    nnoremap <Space>b <c-y>
    nnoremap <Space>m <s-M>
    nnoremap <silent> <SPACE><SPACE> zz

    nnoremap n ^
    nnoremap N g_


    " to be able to search for highlighted words
    vnoremap // y/\V<C-R>=escape(@",'/\')<CR><CR>

    " New way to escape 
    inoremap <silent> <c-J> <c-[>
    vnoremap <silent> <c-J> <c-[>

    nnoremap <silent> <c-x> d$

    " To resize window
    nnoremap <silent> <Tab>- :resize -5<CR>
    nnoremap <silent> <Tab>= :resize +5<CR>

    nnoremap <silent> - :vertical resize -5<CR>
    nnoremap <silent> = :vertical resize +5<CR>
